#include<stdio.h>
#include<stdlib.h>
void bubble_sort(int *a,int n)
{
int i,j,temp;
for(i=n-2;i>=0;i--)
{ for(j=0;j<=i;j++)
{ if(*(a+j)>*(a+j+1))
{ temp=*(a+j);
*(a+j)=*(a+j+1);
*(a+j+1)=temp;
}}}}
void merge(int *c, int *a, int *b,int aLen, int bLen)
{
int i = 0, j = 0;
while (i < aLen && j < bLen)
{
if (a[i] < b[j])
{
c[i+j] = a[i];
i++;
}
else
{ c[i+j] = b[j];
j++;
}
}
while (i < aLen)
{ c[i+j] = a[i];
i++;
}
while (j < bLen)
{ c[i+j] = b[j];
j++;
}}
void main()
{
int *a,*b,*c,i,n,m,d;
printf("Enter the size of an array: ");
scanf("%d",&n);
a=(int*)malloc(n*sizeof(int));
printf("Enter the elements of the array: ");
for(i=0;i<=n-1;i++)
{
scanf("%d",(a+i));
}
bubble_sort(a,n);
printf("Enter the size of an array: ");
scanf("%d",&m);
b=(int*)malloc(m*sizeof(int));
printf("Enter the elements of the array: ");
for(i=0;i<=m-1;i++)
{
scanf("%d",(b+i));
}
bubble_sort(b,m);
d=n+m;
c=(int*)malloc(d*sizeof(int));
merge(c,a,b,n,m);
for(i=0;i<d;i++)
printf("%d ",*(c+i));
}
